step_sum <- function(data) {
    sum_steps <- c()
    current_date <- ""
    steps <- data[, "steps"]
    date <- data[, "date"]
    for (i in 1:nrow(data)) {
        if (current_date != date[i]) {
            current_date <- date[i]
            sum_steps[i] <- steps[i]
        } else {
            sum_steps[i] <- sum_steps[i - 1] + steps[i]
        }
    }
    return(sum_steps)
}
data$sum_step <- step_sum(data)
data$ave_step <- data$sum_step/(data$interval + 5)
